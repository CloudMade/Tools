    def initialize(data)
      @instruction = data[0]
      @length = Float(data[1])
      @position = Integer(data[2])
      @time = Integer(data[3])
      @length_caption = data[4]
      @earth_direction = data[5]
      @azimuth = Float(data[6])
      if data.length == 9
        @turn_type = data[7]
        @turn_angle = data[8]
      end
    end
